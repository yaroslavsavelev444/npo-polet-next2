# /etc/nginx/sites-available/npo-polet.ru.conf
#
# ПЕРЕКЛЮЧЕНИЕ ОСНОВНОГО ДОМЕНА (test.npo-polet.ru -> npo-polet.ru).
#
# ВАЖНО: этот файл сосуществует с ещё активным test.npo-polet.ru.conf до
# самого последнего шага переключения (отключение test. — намеренно
# последнее действие, после проверки, что npo-polet.ru работает). nginx
# хранит имена upstream/map-переменных/limit_req_zone/ssl_session_cache в
# ОБЩЕМ на весь инстанс неймспейсе — двум одновременно загруженным файлам
# нельзя использовать одинаковые имена, даже с идентичным содержимым (первая
# попытка применить этот файл упала именно на этом: "duplicate upstream
# polet_next"). Поэтому ниже везде уникальные, домен-специфичные имена
# (суффикс _main), а не те же, что использует test.npo-polet.ru.conf.
# Отдельный от общего proxy_params_npo.conf инклюд — proxy_params_npo_main.conf
# (см. этот файл рядом) — по той же причине: тот общий файл ссылается на
# map-переменную, объявленную ВНУТРИ test.npo-polet.ru.conf, и перестанет
# существовать вместе с ним на шаге 6.
#
# Перед применением:
# 1. Сверьте с реальным содержимым файла на сервере, если есть сомнения —
#    правки nginx никогда не попадали туда через git/CI (deploy.sh nginx не
#    трогает).
# 2. Сертификат для npo-polet.ru получен заново (certbot certonly --webroot),
#    путь ниже уже правильный: /etc/letsencrypt/live/npo-polet.ru/.
# 3. Скопировать также proxy_params_npo_main.conf в /etc/nginx/.
# 4. Файл в sites-available можно оставить под этим (новым) именем и
#    засимлинковать в sites-enabled; symlink на test.npo-polet.ru.conf
#    убрать ПОСЛЕДНИМ шагом (контейнеры при этом не трогаем — nginx просто
#    перестаёт на этот хост что-либо проксировать).
# 5. `sudo nginx -t` ДО `systemctl reload nginx`. Именно reload, не restart —
#    не роняет уже открытые соединения и не требует ничего делать с Docker.

upstream polet_next_main {
    server 127.0.0.1:3004;
    keepalive 32;
}

map $http_upgrade $connection_upgrade_npo_main {
    default upgrade;
    ''      close;
}

# limit_req_zone обязана быть на уровне http{}, а не внутри server{} (см.
# ngx_http_limit_req_module) — файлы sites-enabled/* инклюдятся прямо в
# http{} главного nginx.conf, поэтому достаточно вынести директиву на
# уровень файла, как здесь.
limit_req_zone $binary_remote_addr zone=npo_auth_main:10m rate=10r/m;

# ── HTTP: acme-challenge + редирект на HTTPS-версию голого домена ──────────
# Оба имени (голое и www) в одном блоке: любой HTTP-запрос, независимо от
# того, каким хостом пришёл, одним 301 уходит сразу на https://npo-polet.ru —
# без промежуточного прыжка через https://www.
server {
    listen 80;
    listen [::]:80;
    server_name npo-polet.ru www.npo-polet.ru;

    location /.well-known/acme-challenge/ {
        root /var/www/certbot;
    }

    location / {
        return 301 https://npo-polet.ru$request_uri;
    }
}

# ── HTTPS: www -> редирект на голый домен ──────────────────────────────────
# Отдельный блок, а не `if` внутри общего server{} (в nginx `if` в контексте
# server/location — известный источник трудноуловимых багов, отдельный
# server{} на server_name — стандартный безопасный способ).
server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name www.npo-polet.ru;

    ssl_certificate     /etc/letsencrypt/live/npo-polet.ru/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/npo-polet.ru/privkey.pem;

    return 301 https://npo-polet.ru$request_uri;
}

# ── HTTPS: основной сервер приложения ──────────────────────────────────────
server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name npo-polet.ru;

    ssl_certificate     /etc/letsencrypt/live/npo-polet.ru/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/npo-polet.ru/privkey.pem;

    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers HIGH:!aNULL:!MD5;
    ssl_session_cache shared:SSL_npo_main:10m;
    ssl_session_timeout 1d;
    ssl_stapling on;
    ssl_stapling_verify on;

    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header Referrer-Policy "strict-origin-when-cross-origin" always;
    add_header Permissions-Policy "camera=(), microphone=(), geolocation=()" always;

    client_max_body_size 25m;

    gzip on;
    gzip_comp_level 5;
    gzip_min_length 512;
    gzip_types
        text/plain
        text/css
        application/json
        application/javascript
        text/xml
        application/xml
        image/svg+xml;

    # Защита login endpoint от брутфорса
    location ~* ^/api/(admins|users)/login {
        limit_req zone=npo_auth_main burst=5 nodelay;
        proxy_pass http://polet_next_main;
        include /etc/nginx/proxy_params_npo_main.conf;
    }

    location ~* ^/(_next/static)/ {
        proxy_pass http://polet_next_main;
        proxy_set_header Host $host;
        add_header Cache-Control "public, max-age=31536000, immutable";
    }

    location /media/ {
        proxy_pass http://polet_next_main;
        proxy_set_header Host $host;
        add_header Cache-Control "public, max-age=604800";
    }

    location / {
        proxy_pass http://polet_next_main;
        include /etc/nginx/proxy_params_npo_main.conf;
    }

    access_log /var/log/nginx/polet-next.access.log;
    error_log  /var/log/nginx/polet-next.error.log;
}
